// Replace txtBarcodeScanner_2.OnChange with this formula.
UpdateContext({varRawScan: Trim(Self.Value)});
If(
    IsBlank(varRawScan),
    Blank(),
    UpdateContext({
        varParsedTracking: If(StartsWith(varRawScan, "PKG|"), Last(Split(varRawScan, "|")).Value, varRawScan),
        varDetectedCarrier: If(
            StartsWith(Lower(varRawScan), "1z"), "UPS",
            IsMatch(varRawScan, "^\\d{12}$|^\\d{15}$|^\\d{20}$|^\\d{34}$"), "FedEx",
            StartsWith(varRawScan, "92") || StartsWith(varRawScan, "93") || StartsWith(varRawScan, "94"), "USPS",
            IsMatch(varRawScan, "^\\d{10}$"), "DHL",
            "Other"
        )
    });
    If(
        IsBlank(Trim(txtRecipient.Value)),
        Notify("Enter a designated recipient before scanning.", NotificationType.Warning),
        IsBlank(varParsedTracking),
        Notify("No tracking number was found in the scan.", NotificationType.Error),
        With(
            {existing: LookUp(Tracking, Title = varParsedTracking && field_4 = Trim(txtRecipient.Value) && field_3 <> "Picked Up")},
            If(
                !IsBlank(existing),
                Notify("This package is already assigned to " & Trim(txtRecipient.Value) & ".", NotificationType.Warning),
                IfError(
                    Set(varSavedTracking, Patch(Tracking, Defaults(Tracking), {
                        Title: varParsedTracking,
                        field_1: varDetectedCarrier,
                        field_2: varScanDirection,
                        field_3: "Ready for Pickup",
                        field_4: Trim(txtRecipient.Value),
                        field_6: Now(),
                        field_7: "Received by commercial tracking scanner"
                    }));
                    Collect(colRecentScans, {
                        TrackingId: varSavedTracking.ID,
                        TrackingNumber: varParsedTracking,
                        Recipient: Trim(txtRecipient.Value),
                        Carrier: varDetectedCarrier,
                        Timestamp: Now(),
                        Direction: varScanDirection
                    });
                    Notify("Package assigned to " & Trim(txtRecipient.Value) & ".", NotificationType.Success),
                    Notify("The package could not be saved: " & FirstError.Message, NotificationType.Error)
                )
            )
        )
    );
    Reset(txtBarcodeScanner_2);
    UpdateContext({varTriggerFocus: true})
)
